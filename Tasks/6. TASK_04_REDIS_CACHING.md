   - **Payment completed** ‚Üí Invalidate appointment, dashboard stats

3. **Cache Strategies**
   - **Read-Heavy Data**: Cache-Aside pattern (lazy loading)
   - **Frequently Updated**: Short TTL with active invalidation
   - **Expensive Queries**: Long TTL with manual invalidation
   - **Session Data**: Store in Redis with auto-expiry

```typescript
import Redis from "ioredis";
import config from "../config";

class RedisClient {
  private client: Redis;
  private static instance: RedisClient;

  private constructor() {
    // Upstash Redis connection
    if (config.redis.url) {
      this.client = new Redis(config.redis.url, {
        // For Upstash, use TLS
        tls: config.env === "production" ? {} : undefined,
        // Retry strategy
        retryStrategy: (times) => {
          const delay = Math.min(times * 50, 2000);
          return delay;
        },
        // Connection timeout
        connectTimeout: 10000,
        // Enable offline queue
        enableOfflineQueue: false,
      });
    } else {
      // Traditional Redis connection
      this.client = new Redis({
        host: config.redis.host,
        port: config.redis.port,
        password: config.redis.password,
        retryStrategy: (times) => {
          const delay = Math.min(times * 50, 2000);
          return delay;
        },
      });
    }

    // Event handlers
    this.client.on("connect", () => {
      console.log("‚úÖ Redis connected successfully");
    });

    this.client.on("error", (error) => {
      console.error("‚ùå Redis connection error:", error);
    });

    this.client.on("ready", () => {
      console.log("‚úÖ Redis ready to accept commands");
    });

    this.client.on("reconnecting", () => {
      console.log("üîÑ Redis reconnecting...");
    });
  }

  public static getInstance(): RedisClient {
    if (!RedisClient.instance) {
      RedisClient.instance = new RedisClient();
    }
    return RedisClient.instance;
  }

  public getClient(): Redis {
    return this.client;
  }

  // Utility methods
  public async get<T>(key: string): Promise<T | null> {
    try {
      const value = await this.client.get(key);
      if (!value) return null;
      return JSON.parse(value) as T;
    } catch (error) {
      console.error("Redis GET error:", error);
      return null;
    }
  }

  public async set(
    key: string,
    value: any,
    ttlSeconds?: number
  ): Promise<boolean> {
    try {
      const stringValue = JSON.stringify(value);
      if (ttlSeconds) {
        await this.client.setex(key, ttlSeconds, stringValue);
      } else {
        await this.client.set(key, stringValue);
      }
      return true;
    } catch (error) {
      console.error("Redis SET error:", error);
      return false;
    }
  }

  public async del(key: string | string[]): Promise<number> {
    try {
      return await this.client.del(key);
    } catch (error) {
      console.error("Redis DEL error:", error);
      return 0;
    }
  }

  public async exists(key: string): Promise<boolean> {
    try {
      const result = await this.client.exists(key);
      return result === 1;
    } catch (error) {
      console.error("Redis EXISTS error:", error);
      return false;
    }
  }

  public async invalidatePattern(pattern: string): Promise<number> {
    try {
      const keys = await this.client.keys(pattern);
      if (keys.length === 0) return 0;
      return await this.client.del(...keys);
    } catch (error) {
      console.error("Redis invalidatePattern error:", error);
      return 0;
    }
  }

  public async increment(key: string): Promise<number> {
    try {
      return await this.client.incr(key);
    } catch (error) {
      console.error("Redis INCR error:", error);
      return 0;
    }
  }

  public async decrement(key: string): Promise<number> {
    try {
      return await this.client.decr(key);
    } catch (error) {
      console.error("Redis DECR error:", error);
      return 0;
    }
  }

  // Hash operations
  public async hset(key: string, field: string, value: any): Promise<number> {
    try {
      return await this.client.hset(key, field, JSON.stringify(value));
    } catch (error) {
      console.error("Redis HSET error:", error);
      return 0;
    }
  }

  public async hget<T>(key: string, field: string): Promise<T | null> {
    try {
      const value = await this.client.hget(key, field);
      if (!value) return null;
      return JSON.parse(value) as T;
    } catch (error) {
      console.error("Redis HGET error:", error);
      return null;
    }
  }

  public async hgetall<T>(key: string): Promise<Record<string, T> | null> {
    try {
      const data = await this.client.hgetall(key);
      if (!data || Object.keys(data).length === 0) return null;

      const parsed: Record<string, T> = {};
      for (const [field, value] of Object.entries(data)) {
        parsed[field] = JSON.parse(value) as T;
      }
      return parsed;
    } catch (error) {
      console.error("Redis HGETALL error:", error);
      return null;
    }
  }

  // List operations
  public async lpush(key: string, ...values: any[]): Promise<number> {
    try {
      const stringValues = values.map((v) => JSON.stringify(v));
      return await this.client.lpush(key, ...stringValues);
    } catch (error) {
      console.error("Redis LPUSH error:", error);
      return 0;
    }
  }

  public async lrange<T>(
    key: string,
    start: number,
    stop: number
  ): Promise<T[]> {
    try {
      const values = await this.client.lrange(key, start, stop);
      return values.map((v) => JSON.parse(v) as T);
    } catch (error) {
      console.error("Redis LRANGE error:", error);
      return [];
    }
  }

  // TTL operations
  public async ttl(key: string): Promise<number> {
    try {
      return await this.client.ttl(key);
    } catch (error) {
      console.error("Redis TTL error:", error);
      return -1;
    }
  }

  public async expire(key: string, seconds: number): Promise<boolean> {
    try {
      const result = await this.client.expire(key, seconds);
      return result === 1;
    } catch (error) {
      console.error("Redis EXPIRE error:", error);
      return false;
    }
  }

  // Connection management
  public async disconnect(): Promise<void> {
    await this.client.quit();
  }

  public async flushdb(): Promise<void> {
    await this.client.flushdb();
  }
}
```

```typescript
import crypto from "crypto";
import { redisClient } from "../lib/redis";

export class CacheService {
  private static prefix = "ph-healthcare";

  // Generate cache key
  private static generateKey(...parts: (string | number)[]): string {
    return `${this.prefix}:${parts.join(":")}`;
  }

  // Generate hash for complex objects (filters, queries)
  private static hash(obj: any): string {
    return crypto
      .createHash("md5")
      .update(JSON.stringify(obj))
      .digest("hex")
      .substring(0, 8);
  }

  // ===== DOCTOR CACHING =====

  static async cacheDoctorList(
    filters: any,
    data: any,
    ttl = 300
  ): Promise<void> {
    const filterHash = this.hash(filters);
    const key = this.generateKey("doctors", "list", filterHash);
    await redisClient.set(key, data, ttl);
  }

  static async getCachedDoctorList(filters: any): Promise<any | null> {
    const filterHash = this.hash(filters);
    const key = this.generateKey("doctors", "list", filterHash);
    return await redisClient.get(key);
  }

  static async cacheDoctorDetails(
    doctorId: string,
    data: any,
    ttl = 300
  ): Promise<void> {
    const key = this.generateKey("doctors", "details", doctorId);
    await redisClient.set(key, data, ttl);
  }

  static async getCachedDoctorDetails(doctorId: string): Promise<any | null> {
    const key = this.generateKey("doctors", "details", doctorId);
    return await redisClient.get(key);
  }

  static async invalidateDoctorCache(doctorId?: string): Promise<void> {
    if (doctorId) {
      // Invalidate specific doctor
      await redisClient.del(this.generateKey("doctors", "details", doctorId));
      await redisClient.del(this.generateKey("doctors", "stats", doctorId));
    }
    // Invalidate all doctor lists (with different filters)
    await redisClient.invalidatePattern(
      this.generateKey("doctors", "list", "*")
    );
  }

  // ===== SPECIALTY CACHING =====

  static async cacheSpecialtyList(data: any, ttl = 1800): Promise<void> {
    const key = this.generateKey("specialties", "list");
    await redisClient.set(key, data, ttl);
  }

  static async getCachedSpecialtyList(): Promise<any | null> {
    const key = this.generateKey("specialties", "list");
    return await redisClient.get(key);
  }

  static async invalidateSpecialtyCache(): Promise<void> {
    await redisClient.invalidatePattern(this.generateKey("specialties", "*"));
    // Also invalidate doctor lists (they include specialty info)
    await this.invalidateDoctorCache();
  }

  // ===== APPOINTMENT CACHING =====

  static async cacheAppointmentList(
    userId: string,
    role: string,
    data: any,
    ttl = 60
  ): Promise<void> {
    const key = this.generateKey("appointments", role.toLowerCase(), userId);
    await redisClient.set(key, data, ttl);
  }

  static async getCachedAppointmentList(
    userId: string,
    role: string
  ): Promise<any | null> {
    const key = this.generateKey("appointments", role.toLowerCase(), userId);
    return await redisClient.get(key);
  }

  static async invalidateAppointmentCache(
    userId?: string,
    role?: string
  ): Promise<void> {
    if (userId && role) {
      await redisClient.del(
        this.generateKey("appointments", role.toLowerCase(), userId)
      );
    } else {
      // Invalidate all appointments
      await redisClient.invalidatePattern(
        this.generateKey("appointments", "*")
      );
    }
    // Also invalidate dashboard stats
    await this.invalidateDashboardCache();
  }

  // ===== DASHBOARD CACHING =====

  static async cacheDashboardStats(
    role: string,
    userId: string | null,
    data: any,
    ttl = 120
  ): Promise<void> {
    const key = userId
      ? this.generateKey("dashboard", role.toLowerCase(), userId, "stats")
      : this.generateKey("dashboard", role.toLowerCase(), "stats");
    await redisClient.set(key, data, ttl);
  }

  static async getCachedDashboardStats(
    role: string,
    userId: string | null
  ): Promise<any | null> {
    const key = userId
      ? this.generateKey("dashboard", role.toLowerCase(), userId, "stats")
      : this.generateKey("dashboard", role.toLowerCase(), "stats");
    return await redisClient.get(key);
  }

  static async invalidateDashboardCache(
    role?: string,
    userId?: string
  ): Promise<void> {
    if (role && userId) {
      await redisClient.del(
        this.generateKey("dashboard", role.toLowerCase(), userId, "stats")
      );
    } else if (role) {
      await redisClient.invalidatePattern(
        this.generateKey("dashboard", role.toLowerCase(), "*")
      );
    } else {
      await redisClient.invalidatePattern(this.generateKey("dashboard", "*"));
    }
  }

  // ===== AI QUERY CACHING =====

  static async cacheAIQuery(
    query: string,
    response: any,
    ttl = 86400
  ): Promise<void> {
    const queryHash = this.hash(query.toLowerCase().trim());
    const key = this.generateKey("ai", "query", queryHash);
    await redisClient.set(key, response, ttl);
  }

  static async getCachedAIQuery(query: string): Promise<any | null> {
    const queryHash = this.hash(query.toLowerCase().trim());
    const key = this.generateKey("ai", "query", queryHash);
    return await redisClient.get(key);
  }

  // ===== SESSION CACHING =====

  static async cacheSession(
    userId: string,
    sessionId: string,
    data: any,
    ttl = 3600
  ): Promise<void> {
    const key = this.generateKey("sessions", "user", userId, sessionId);
    await redisClient.set(key, data, ttl);
  }

  static async getCachedSession(
    userId: string,
    sessionId: string
  ): Promise<any | null> {
    const key = this.generateKey("sessions", "user", userId, sessionId);
    return await redisClient.get(key);
  }

  static async invalidateUserSessions(userId: string): Promise<void> {
    await redisClient.invalidatePattern(
      this.generateKey("sessions", "user", userId, "*")
    );
  }

  // ===== RATE LIMITING =====

  static async incrementRateLimit(
    identifier: string,
    windowSeconds: number
  ): Promise<number> {
    const key = this.generateKey("rate-limit", identifier);
    const count = await redisClient.increment(key);

    // Set expiry on first request
    if (count === 1) {
      await redisClient.expire(key, windowSeconds);
    }

    return count;
  }

  static async getRateLimitCount(identifier: string): Promise<number> {
    const key = this.generateKey("rate-limit", identifier);
    const value = await redisClient.get<number>(key);
    return value || 0;
  }

  // ===== CACHE STATISTICS =====

  static async recordCacheHit(cacheType: string): Promise<void> {
    const key = this.generateKey("stats", "hits", cacheType);
    await redisClient.increment(key);
  }

  static async recordCacheMiss(cacheType: string): Promise<void> {
    const key = this.generateKey("stats", "misses", cacheType);
    await redisClient.increment(key);
  }

  static async getCacheStats(): Promise<any> {
    const hitKeys = await redisClient
      .getClient()
      .keys(this.generateKey("stats", "hits", "*"));
    const missKeys = await redisClient
      .getClient()
      .keys(this.generateKey("stats", "misses", "*"));

    const stats: Record<string, any> = {};

    for (const key of hitKeys) {
      const type = key.split(":").pop() || "unknown";
      const hits = (await redisClient.get<number>(key)) || 0;
      stats[type] = { ...(stats[type] || {}), hits };
    }

    for (const key of missKeys) {
      const type = key.split(":").pop() || "unknown";
      const misses = (await redisClient.get<number>(key)) || 0;
      stats[type] = { ...(stats[type] || {}), misses };
    }

    // Calculate hit rates
    for (const type in stats) {
      const { hits = 0, misses = 0 } = stats[type];
      const total = hits + misses;
      stats[type].hitRate = total > 0 ? (hits / total) * 100 : 0;
    }

    return stats;
  }
}
```
